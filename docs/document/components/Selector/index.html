Selector 选择器 #

当选项过多时，使用下拉菜单展示并选择内容

  • 可以根据输入框内容进行选项筛选
  • 可以使用上下方向键移动选择光标
  • 可以使用回车选定


  • Vue.js
  • React.js
  • Angular.js
  • jQuery.js
<template>
<div class="myContainer">
    <lb-selector 
    :data="selectorData"
    :selected="selected"
    placeholder="请选择框架"
    class="selectorBox"
    ></lb-selector>
</div>
</template>

<script>
export default{
    setup(){
        const selectorData = [
            {
                id:1,
                value:'Vue.js'
            },
            {
                id:1,
                value:'React.js'
            },
            {
                id:1,
                value:'Angular.js'
            },
            {
                id:1,
                value:'jQuery.js'
            }
        ];
        function selected(value){
            alert(value)
        }
        return {
            selectorData,
            selected
        }
    }
}
</script>
<style scoped>
.myContainer{
    height:30px;
}
.myContainer .selectorBox{
    position:absolute;
    z-index:2;
}
</style>
